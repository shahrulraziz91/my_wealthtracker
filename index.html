function subscribeUsers(showLanding){ const {ref, onValue}=window.__FB; const usersRef=ref(db,'users'); if(unsubscribeUsers) unsubscribeUsers(); let firstLoad=true; unsubscribeUsers=onValue(usersRef, snap=>{ users={}; snap.forEach(child=>{ users[child.key]=child.val(); }); hideLoading(); setFbStatus('connected','🔥 Firebase Connected'); if(firstLoad){ // First time — decide which screen to show firstLoad=false; if(isViewMode && viewingUserId && users[viewingUserId]){ enterViewMode(viewingUserId); } else { showScreen('screenLanding'); renderLanding(); } } else { // Subsequent updates — silently refresh without changing screen if(currentUser){ currentUser=users[currentUser.id]||currentUser; } // Refresh dashboard if active if(document.getElementById('screenDashboard').classList.contains('active')){ renderAll(); } // Refresh landing cards if active if(document.getElementById('screenLanding').classList.contains('active')){ renderLanding(); } } }, err=>{ setFbStatus('error','❌ Connection Error'); hideLoading(); showToast('❌ Firebase error: '+err.message); }); }
